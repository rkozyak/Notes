\section{Position, Velocity, and Acceleration Vectors in Multiple Dimensions}

% Derivates still work
\subsection{Derivates still work}
\[v=\frac{dr}{dt}\]
\[\frac{d}{dt}(xi+yj+zk)\]
\[\frac{dx}{dt}i + \frac{dy}{dt} J + \frac{dz}{dt} k\]
\[v_x i +v_uj+v_zk\]

% Kinematic Equations
\subsection{Kinematics Equations}
\[v=v_o+at\]
\[r=r_i+v_ot+\frac{1}{2}at^2\]
\[v^2=v_o^2+2a\Delta r\]

% 1 Dimensions
\begin{center}
1 Dimension: \\
x = position 
$\Delta x$: displacement
v:velocity
a: acceleration
\end{center}

% 2 or 3 Dimensions
\begin{center} 
2 or 3 Dimensions: \\
r: position 
$\Delta r$ + displacement
v: velocity
a: acceleration 
\end{center}


% Unit Vector Notation
\subsection{Unit Vector Notation}
\[r=xi+yj+zk\]
\[\Delta r= \Delta xi+\Delta yj+\Delta zk\]
\[v=v_xi+v_y+j+v_zk\]
\[a=a_xi+a_yj+a_zk\]
